69def parse_log(line: str) -> dict: 

70 """Try to parse the log line with some predefined format 

71 

72 Currently only support three formats: 

73 - klog 

74 - logr 

75 - json 

76 

77 Returns: 

78 a dict containing 'level' and 'message' 

79 'level' will always be a lowercase string 

80 """ 

81 logger = get_thread_logger(with_prefix=True) 

82 

83 log_line = {} 

84 if re.search(KLOG_REGEX, line) is not None: 

85 # log is in klog format 

86 match = re.search(KLOG_REGEX, line) 

87 if match is None: 

88 logger.debug("parse_log() cannot parse line %s", line) 

89 return {} 

90 if match.group(1) == "E": 

91 log_line["level"] = "error" 

92 elif match.group(1) == "I": 

93 log_line["level"] = "info" 

94 elif match.group(1) == "W": 

95 log_line["level"] = "warn" 

96 elif match.group(1) == "F": 

97 log_line["level"] = "fatal" 

98 

99 log_line["msg"] = match.group(11) 

100 elif re.search(LOGR_REGEX, line) is not None: 

101 # log is in logr format 

102 match = re.search(LOGR_REGEX, line) 

103 if match is None: 

104 logger.debug("parse_log() cannot parse line %s", line) 

105 return {} 

106 log_line["level"] = match.group(2).lower() 

107 log_line["msg"] = match.group(4) 

108 elif re.search(LOGR_SPECIAL_REGEX, line) is not None: 

109 # log is in logr special format 

110 match = re.search(LOGR_SPECIAL_REGEX, line) 

111 if match is None: 

112 logger.debug("parse_log() cannot parse line %s", line) 

113 return {} 

114 log_line["level"] = match.group(2).lower() 

115 log_line["msg"] = match.group(4) 

116 elif re.search(LOGRUS_REGEX, line) is not None: 

117 # log is in logrus format 

118 match = re.search(LOGRUS_REGEX, line) 

119 if match is None: 

120 logger.debug("parse_log() cannot parse line %s", line) 

121 return {} 

122 log_line["level"] = match.group(2) 

123 log_line["msg"] = match.group(3) 

124 elif re.search(GRAFANA_LOGR_REGEX, line) is not None: 

125 match = re.search(GRAFANA_LOGR_REGEX, line) 

126 if match is None: 

127 logger.debug("parse_log() cannot parse line %s", line) 

128 return {} 

129 log_line["level"] = match.group(2).lower() 

130 log_line["msg"] = match.group(4) 

131 elif re.search(KAFKA_LOG_REGEX, line) is not None: 

132 match = re.search(KAFKA_LOG_REGEX, line) 

133 if match is None: 

134 logger.debug("parse_log() cannot parse line %s", line) 

135 return {} 

136 log_line["level"] = match.group(2).lower() 

137 log_line["msg"] = match.group(5)

147 logger.debug("parse_log() cannot parse line %s due to %s", line, e)
